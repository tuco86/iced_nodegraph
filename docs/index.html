üì¶ Quick Start

Add iced_nodegraph to your Cargo.toml and start building!

[dependencies] iced = { git = "https://github.com/iced-rs/iced.git", branch = "master" } iced_nodegraph = { git = "https://github.com/tuco86/iced_nodegraph.git" }
use iced_nodegraph::{node_graph, node_pin, PinDirection, PinSide}; // Create a node graph widget let mut graph = node_graph() .on_connect(|from_node, from_pin, to_node, to_pin| { Message::EdgeConnected(from_node, from_pin, to_node, to_pin) }) .on_move(|node_id, position| { Message::NodeMoved(node_id, position) }); // Add nodes with custom content graph = graph.push_node( Point::new(100.0, 100.0), container(column![ text("Process Node"), node_pin("Input", PinDirection::Input, PinSide::Left), node_pin("Output", PinDirection::Output, PinSide::Right), ]) );
